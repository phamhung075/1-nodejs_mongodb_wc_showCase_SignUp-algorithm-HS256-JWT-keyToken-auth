``` js
'use strict'

const JWT = require ('jsonwebtoken');

const createTokenPair = async ( payload, cryptoKey ) => {// payload mang theo khối lượng gì đó để ta mã hóa đẩy vào token -> tạo ra set token và request token
    try{
        //accessToken
		const accessToken = await JWT.sign ( payload, cryptoKey, { //algorithm HS256 par default
        
        //payload chứa thông tin vận chuyển mang từ chỗ này sang chỗ khác
            expiresIn: '2 days'
        }); 
        //payload: chua thong tin van chuyen mang di tu he thong nay sang he thong khac thong qua token
        //privateKey: khong luu vao database chi xay ra, diễn ra 1 lần khi sign hoặc login thành công -> đẩy qua browser
        // nếu đẩy qua browser ta nhìn thấy thì sao, bản thân ta nhìn thấy token của mình là điều bình thường, session facebook ta mở lên cũng thấy session cookie id của ta, quan trọng là người khác có lấy được token của ta hay k
        const refreshToken = await JWT.sign ( payload, cryptoKey, {
            expiresIn: '7 days'
        }); 


//hàm verify sử dung public token, nếu như bình thường sử dụng một keysecret vừa sign vừa verify là một sai lầm phổ biến nhất, người ta bắt được key secret của ta, người ta có thể verify chữ kí đồng thời tạo ra chữ kí của chúng ta luôn.
        JWT.verify ( accessToken, cryptoKey, (err, decode) => {
            if(err){
                console.error(`error verify::`, err);
            }else{
                console.log(`decode verify::`, decode);
            }
        }); 
        return { accessToken, refreshToken }
    }catch (error) {
		return error
    }
}

module.exports = {
    createTokenPair
}
```
